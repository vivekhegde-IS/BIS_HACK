inference.py — root level — Person A writes

This is the entry point judges run. It must accept --input and --output flags. It loads the saved index, processes every query, and writes results to a JSON file.

import argparse, json, time, sys
sys.path.insert(0, "src")
from retriever import load_index, retrieve

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input",  required=True)
    parser.add_argument("--output", required=True)
    args = parser.parse_args()

    index = load_index("data/tfidf_model.pkl")  # load ONCE, not inside loop

    with open(args.input) as f:
        queries = json.load(f)

    results = []
    for item in queries:
        t0 = time.time()
        ids = retrieve(item["query"], index, top_k=5)
        elapsed = time.time() - t0

        while len(ids) < 5:          # always pad to exactly 5
            ids.append(ids[-1])

        results.append({
            "id":                  item["id"],
            "retrieved_standards": ids[:5],
            "latency_seconds":     round(elapsed, 4)
        })

    with open(args.output, "w") as f:
        json.dump(results, f, indent=2)
    print(f"Done: {len(results)} queries processed")

if __name__ == "__main__":
    main()
Output keys must be exactly: id, retrieved_standards, latency_seconds — not "standards", not "results", not anything else. Wrong keys = eval script crashes = 0/40 pts.

src/ingest.py — Person A writes

Reads BIS_SP_21.txt (converted from PDF using pdftotext) and produces data/chunks.json. Each chunk is one IS standard entry.

import re, json

def parse_standards(txt_path):
    with open(txt_path, 'r', encoding='utf-8', errors='ignore') as f:
        text = f.read()
    pattern = r'(IS\s+\d+[\w\s:.-]*?:\s*\d{4})'
    parts = re.split(pattern, text)
    chunks = []
    i = 1
    while i < len(parts) - 1:
        is_id = parts[i].strip()
        body  = parts[i+1].strip()[:1500]
        title_match = re.match(r'([A-Z][A-Z\s/,-]{5,80})', body)
        title = title_match.group(1).strip() if title_match else is_id
        chunks.append({
            "id":    is_id,
            "title": title,
            "text":  f"{is_id} {title} {body}"
        })
        i += 2
    return chunks

if __name__ == "__main__":
    chunks = parse_standards("BIS_SP_21.txt")
    with open("data/chunks.json", "w") as f:
        json.dump(chunks, f, indent=2)
    print(f"Extracted {len(chunks)} chunks")

src/indexer.py — Person A writes

Reads chunks.json, builds a TF-IDF model using scikit-learn, and saves the model plus the chunk list to data/tfidf_model.pkl so inference.py can load it instantly.

import json, pickle
from sklearn.feature_extraction.text import TfidfVectorizer

def build_index(chunks_path="data/chunks.json",
                out_path="data/tfidf_model.pkl"):
    with open(chunks_path) as f:
        chunks = json.load(f)
    texts = [c["text"] for c in chunks]
    vec = TfidfVectorizer(ngram_range=(1,2), min_df=1, sublinear_tf=True)
    mat = vec.fit_transform(texts)
    with open(out_path, "wb") as f:
        pickle.dump({"vectorizer": vec, "matrix": mat, "chunks": chunks}, f)
    print(f"Index: {mat.shape[0]} docs, {mat.shape[1]} terms")

if __name__ == "__main__":
    build_index()

src/retriever.py — Person A writes

Loads the saved index, expands the query with synonyms, runs cosine similarity, and returns the top-5 IS standard IDs.

import pickle, numpy as np
from sklearn.metrics.pairwise import cosine_similarity

SYNONYMS = {
    "slag":      "portland slag",
    "pozzolana": "fly ash calcined clay",
    "aggregate": "coarse fine gravel",
    "lime":      "building lime quicklime",
    "cement":    "portland ordinary",
    "plywood":   "wood panel board",
    "brick":     "masonry fired clay",
    "pipe":      "tube conduit drainage",
    "steel":     "iron structural metal bar",
    "tile":      "ceramic floor wall",
    "paint":     "coating surface primer",
    "wire":      "conductor cable electrical",
}

def expand_query(q):
    q = q.lower()
    for k, v in SYNONYMS.items():
        if k in q:
            q = q + " " + v
    return q

def load_index(path="data/tfidf_model.pkl"):
    with open(path, "rb") as f:
        return pickle.load(f)

def retrieve(query, index, top_k=5):
    vec = index["vectorizer"].transform([expand_query(query)])
    scores = cosine_similarity(vec, index["matrix"])[0]
    top = np.argsort(scores)[::-1][:top_k]
    return [index["chunks"][i]["id"] for i in top]

src/rationale.py — Person B writes

Calls the Anthropic API to generate a one-sentence explanation of why a retrieved IS standard is relevant to the query. Only used by the Flask UI — not by inference.py.

import os, anthropic
from dotenv import load_dotenv
load_dotenv()

def get_rationale(query: str, standard_id: str) -> str:
    try:
        client = anthropic.Anthropic(
            api_key=os.getenv("ANTHROPIC_API_KEY")
        )
        msg = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=120,
            messages=[{"role": "user", "content":
                f"User searched: '{query}'\n"
                f"Retrieved standard: {standard_id}\n"
                f"In one sentence explain why this IS standard is relevant."
            }]
        )
        return msg.content[0].text
    except Exception as e:
        return f"(Explanation unavailable)"
Never hardcode the API key in this file or anywhere in the codebase. os.getenv reads it from .env which is never pushed to GitHub.

src/app.py — Person B writes

Flask web application. Loads the retriever index, accepts a query from the browser, calls retrieve() and get_rationale(), and returns results as JSON to be displayed.

import sys, os, time
sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))
from flask import Flask, request, jsonify, render_template_string
from src.retriever import load_index, retrieve
from src.rationale import get_rationale

app = Flask(__name__)
INDEX = load_index("data/tfidf_model.pkl")

PAGE = """<!DOCTYPE html><html><head><title>BIS Search</title>
<style>body{font-family:sans-serif;max-width:680px;margin:40px auto;padding:0 16px;}
input{width:100%;padding:10px;font-size:15px;border:1px solid #ccc;border-radius:6px;}
button{margin-top:8px;padding:9px 20px;background:#185FA5;color:#fff;border:none;
  border-radius:6px;cursor:pointer;}
.r{border:1px solid #ddd;border-radius:8px;padding:12px;margin:8px 0;}
.rk{font-weight:600;color:#185FA5;}</style></head>
<body><h2>BIS Standards Search</h2>
<input id="q" placeholder="Describe your product...">
<button onclick="go()">Search</button>
<div id="out"></div>
<script>
async function go(){
  const q=document.getElementById('q').value;
  document.getElementById('out').innerHTML='Searching...';
  const r=await fetch('/search',{method:'POST',
    headers:{'Content-Type':'application/json'},
    body:JSON.stringify({query:q})});
  const d=await r.json();
  document.getElementById('out').innerHTML=d.results.map((x,i)=>
    `<div class="r"><span class="rk">#${i+1}</span> ${x.id}
    <p>${x.rationale}</p></div>`
  ).join('')+'<p style="color:#888">Latency: '+d.latency+'s</p>';
}
</script></body></html>"""

@app.route("/")
def home(): return render_template_string(PAGE)

@app.route("/search", methods=["POST"])
def search():
    q = request.json.get("query","")
    t = time.time()
    ids = retrieve(q, INDEX, top_k=5)
    lat = round(time.time()-t, 3)
    results=[{"id":i,"rationale":get_rationale(q,i)} for i in ids]
    return jsonify({"results":results,"latency":lat})

if __name__=="__main__":
    app.run(debug=True, port=5000)

Person A — AI coding rules and git safety rules

Since you are building with AI assistance, these rules stop the AI from silently breaking things while generating code.

Rules about what the AI must NOT do when writing Person A's code

Rule A-AI-1. The AI must never place inference.py inside /src. If it does, immediately move it to root. The judges run: python inference.py from the repo root — not python src/inference.py.
Rule A-AI-2. The AI must never change the output JSON key names. The only acceptable keys are id, retrieved_standards, and latency_seconds. If the AI writes "standards", "results", "retrieved", or anything else, fix it immediately.
